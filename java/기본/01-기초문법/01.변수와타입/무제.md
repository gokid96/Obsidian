package 기본.기초문법.변수와타입;

public class 기본타입 {
    public static void main(String[] args) {
        
        // ========== 정수 타입 ==========
        
        // byte: 1byte (-128 ~ 127)
        byte b = 127;
        // byte b2 = 128;  // 컴파일 에러: 범위 초과
        
        // short: 2byte (-32,768 ~ 32,767)
        short s = 32767;
        
        // int: 4byte (약 -21억 ~ 21억) - 정수 기본 타입
        int i = 2147483647;
        int billion = 1_000_000_000;  // 언더스코어로 가독성 향상 (Java 7+)
        
        // long: 8byte - 접미사 L 필수
        long l = 9223372036854775807L;
        
        System.out.println("=== 정수 타입 ===");
        System.out.println("byte: " + b);
        System.out.println("short: " + s);
        System.out.println("int: " + i);
        System.out.println("int (언더스코어): " + billion);
        System.out.println("long: " + l);
        
        
        // ========== 실수 타입 ==========
        
        // float: 4byte - 접미사 f 필수
        float f = 3.14f;
        
        // double: 8byte - 실수 기본 타입
        double d = 3.141592653589793;
        
        System.out.println("\n=== 실수 타입 ===");
        System.out.println("float: " + f);
        System.out.println("double: " + d);
        
        // 정밀도 차이
        System.out.println("\n=== 정밀도 차이 ===");
        float f2 = 0.1f;
        double d2 = 0.1;
        System.out.println("float 0.1: " + f2);   // 부정확
        System.out.println("double 0.1: " + d2);  // 더 정확
        
        
        // ========== 문자 타입 ==========
        
        // char: 2byte (0 ~ 65,535) - 유니코드
        char c1 = 'A';
        char c2 = '가';
        char c3 = 65;       // 숫자로도 가능 (A의 유니코드)
        char c4 = '\u0041'; // 유니코드 직접 입력 (A)
        
        System.out.println("\n=== 문자 타입 ===");
        System.out.println("char 'A': " + c1);
        System.out.println("char '가': " + c2);
        System.out.println("char 65: " + c3);
        System.out.println("char \\u0041: " + c4);
        System.out.println("'A'의 유니코드: " + (int) c1);
        
        
        // ========== 논리 타입 ==========
        
        // boolean: true / false
        boolean bool1 = true;
        boolean bool2 = false;
        boolean bool3 = (10 > 5);  // 조건식 결과
        
        System.out.println("\n=== 논리 타입 ===");
        System.out.println("boolean true: " + bool1);
        System.out.println("boolean false: " + bool2);
        System.out.println("10 > 5: " + bool3);
        
        
        // ========== 기본값 확인 ==========
        // 멤버 변수는 기본값으로 초기화됨
        // 지역 변수는 반드시 초기화 필요
        
        System.out.println("\n=== 기본값 (DefaultValues 클래스 참고) ===");
        DefaultValues dv = new DefaultValues();
        dv.printDefaults();
    }
}

// 기본값 확인용 클래스
class DefaultValues {
    byte b;
    short s;
    int i;
    long l;
    float f;
    double d;
    char c;
    boolean bool;
    
    void printDefaults() {
        System.out.println("byte 기본값: " + b);
        System.out.println("short 기본값: " + s);
        System.out.println("int 기본값: " + i);
        System.out.println("long 기본값: " + l);
        System.out.println("float 기본값: " + f);
        System.out.println("double 기본값: " + d);
        System.out.println("char 기본값: [" + c + "] (유니코드 " + (int) c + ")");
        System.out.println("boolean 기본값: " + bool);
    }
}
