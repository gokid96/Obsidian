// 다대일 (가장 기본)
@Entity
public class Member {
    @Id @GeneratedValue
    private Long id;
    
    @ManyToOne  // 다대일
    @JoinColumn(name = "team_id")
    private Team team;
}

@Entity
public class Team {
    @Id @GeneratedValue
    private Long id;
    
    private String name;
}

// 사용
Team team = new Team();
em.persist(team);
