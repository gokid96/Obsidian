```java
@Entity  // JPA가 관리
@Table(name = "users")
public class User {
    
    @Id  // 기본키
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;
    
    @Column(name = "user_name", nullable = false, length = 50)
    private String name;
    
    private int age;  // @Column 생략 가능
    
    // 기본 생성자 필수
    protected User() {}
    
    // Getter, Setter
}
```
