### 코드
``` java
import java.util.Scanner;

public class Main{
    public static void main (String[] args){
        Scanner sc = new Scanner(System.in);
        int T = sc.nextInt();
        for(int i=0; i<T; i++){
          int k = sc.nextInt(); //2
          int n = sc.nextInt(); //3
			//0층 [1,2,3]
	        //1층 [1,3,6]
	        //2층 [1,4,10]
          int [] floor = new int [n];
          
          for(int y=0; y<n; y++){
              floor[y] += y+1; 
          }
          
          for(int j=1; j<=k; j++){//1~k층
            for (int q=1; q<n; q++){ //1~n호
                floor[q] += floor[q-1]; 
            }
          }
          System.out.println(floor[n-1]);// 10
          }    
    }
```
